    const totalCountEl = document.getElementById("totalCount");

    function renderCountries(list) {
      container.innerHTML = "";
      list.forEach(c => {
        const card = document.createElement("div");
        card.className = "country-card";
        card.textContent = c.toUpperCase();
        container.appendChild(card);
      });
    }

    function filterCountries(keyword) {
      let filtered;
      if (searchMode === "start") {
        filtered = countries.filter(c => c.toLowerCase().startsWith(keyword.toLowerCase()));
      } else {
        filtered = countries.filter(c => c.toLowerCase().includes(keyword.toLowerCase()));
      }
      renderCountries(filtered);
      if (keyword) {
        searchInfo.innerHTML = `Countries ${searchMode === "start" ? "starting with" : "containing"} <span class="highlight">${keyword.toUpperCase()}</span> are ${filtered.length}`;
      } else {
        searchInfo.textContent = "";
      }
    }

    input.addEventListener("input", e => filterCountries(e.target.value));
    btnStart.addEventListener("click", () => {
      searchMode = "start";
      filterCountries(input.value);
    });
    btnAny.addEventListener("click", () => {
      searchMode = "any";
      filterCountries(input.value);
    });
    btnSort.addEventListener("click", () => {
      countries.sort((a, b) => sortAsc ? a.localeCompare(b) : b.localeCompare(a));
      sortAsc = !sortAsc;
      filterCountries(input.value);
    });

    renderCountries(countries);
